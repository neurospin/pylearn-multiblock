14 -def A_from_groups(num_variables, groups, weights=None): 

15      """Generates the linear operator for the group lasso Nesterov function 
16      from the groups of variables. 
17   
18      Parameters: 
19      ---------- 
20      num_variables : Integer. The total number of variables. 
21   
22      groups : A list of lists. The outer list represents the groups and the 
23              inner lists represent the variables in the groups. E.g. [[1, 2], 
24              [2, 3]] contains two groups ([1, 2] and [2, 3]) with variable 1 and 
25              2 in the first group and variables 2 and 3 in the second group. 
26   
27      weights : A list. Weights put on the groups. Default is weight 1 for each 
28              group. 
29      """ 
30      if weights is None: 
31          weights = [1.0] * len(groups) 
32   
33      A = list() 
34      for g in xrange(len(groups)): 
35          Gi = groups[g] 
36          lenGi = len(Gi) 
37          Ag = sparse.lil_matrix((lenGi, num_variables)) 
38          for i in xrange(lenGi): 
39              w = weights[g] 
40              Ag[i, Gi[i]] = w 
41   
42          # Matrix operations are a lot faster when the sparse matrix is csr 
43          A.append(Ag.tocsr()) 
44   
45      return A
